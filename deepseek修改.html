    <script>

        const canvas = document.getElementById("wheel");

        const ctx = canvas.getContext("2d");

        const spinButton = document.getElementById("spin-button");

        const resultText = document.getElementById("result");


        const keywords = [

            "🎊 靈魂訊息解鎖", "🌀 覺察", "💫 釋放", "🔥 蛻變", "💡 神秘折扣碼", "🌱 信任", "🚀 行動", "🎟️ 限時優惠", 

            "🌊 流動", "⚖️ 穩定", "🌟 突破", "💖 愛", "🎯 專注", "🎟️ 一對一直覺訊息", "🧧紅包機會"

        ];

        const colors = [

            "#345E7D", "#6DA7CC", "#88A9C3", "#345E7D", "#6DA7CC", "#88A9C3", "#345E7D", "#6DA7CC", 

            "#88A9C3", "#345E7D", "#6DA7CC", "#88A9C3", "#345E7D", "#6DA7CC", "#345E7D"

        ];

        const segmentAngle = (2 * Math.PI) / keywords.length;


        function drawWheel() {

            console.log("Drawing wheel..."); // 調試訊息

            ctx.clearRect(0, 0, canvas.width, canvas.height);


            for (let i = 0; i < keywords.length; i++) {

                ctx.beginPath();

                ctx.moveTo(200, 200);

                ctx.arc(200, 200, 200, i * segmentAngle, (i + 1) * segmentAngle);

                ctx.closePath();

                ctx.fillStyle = colors[i];

                ctx.fill();

                ctx.strokeStyle = "#fff";

                ctx.stroke();


                ctx.save();

                ctx.translate(200, 200);

                ctx.rotate(i * segmentAngle + segmentAngle / 2);

                ctx.fillStyle = "white";

                ctx.font = "14px 'Noto Sans TC', sans-serif";

                ctx.textAlign = "center";

                ctx.fillText(keywords[i], 130, 10);

                ctx.restore();

            }

        }


        let rotation = 0;

        function spinWheel() {

            spinButton.disabled = true;

            const randomDegree = Math.floor(Math.random() * 360) + 1440; // 隨機旋轉角度

            rotation += randomDegree;

            canvas.style.transition = 'transform 3s ease-out';

            canvas.style.transform = `rotate(${rotation}deg)`;


            setTimeout(() => {

                const actualDegree = rotation % 360;

                const correctedDegree = (360 - actualDegree) % 360; // 修正角度計算

                const index = Math.floor(correctedDegree / (360 / keywords.length));

                resultText.innerHTML = `<div style='font-size:24px; font-weight:bold; color:#D32F2F;'>🎉 你抽中了：${keywords[index]}！ 🎉</div>

                    <div style='font-size:18px; margin-top:5px;'>你的新年智慧訊息已準備好，請私訊 @originpaths 領取你的專屬指引與驚喜！ ✨💫</div>`;

                spinButton.disabled = false;

            }, 3000);

        }


        spinButton.addEventListener("click", spinWheel);

        drawWheel(); // 繪製輪盤

    <script>

        const canvas = document.getElementById("wheel");

        const ctx = canvas.getContext("2d");

        const spinButton = document.getElementById("spin-button");

        const resultText = document.getElementById("result");


        const keywords = ["🌀 覺察", "💫 釋放", "🔥 蛻變", "🌱 信任", "🚀 行動", "🌊 流動", "⚖️ 穩定", "🌟 突破", "💖 愛", "🎯 專注", "💡 神秘折扣碼", "🎟️ 限時優惠", "🔢 隱藏占卜機會", "🎊 靈魂訊息解鎖"];

        const colors = ["#345E7D", "#6DA7CC", "#88A9C3", "#345E7D", "#6DA7CC", "#88A9C3", "#345E7D", "#6DA7CC", "#88A9C3", "#345E7D", "#6DA7CC", "#88A9C3", "#345E7D", "#6DA7CC"];

        const segmentAngle = (2 * Math.PI) / keywords.length;


        function drawWheel() {

            for (let i = 0; i < keywords.length; i++) {

                ctx.beginPath();

                ctx.moveTo(100, 100);

                ctx.arc(100, 100, 100, i * segmentAngle, (i + 1) * segmentAngle);

                ctx.closePath();

                ctx.fillStyle = colors[i];

                ctx.fill();

                ctx.strokeStyle = "#fff";

                ctx.stroke();

                

                ctx.save();

                ctx.translate(100, 100);

                ctx.rotate(i * segmentAngle + segmentAngle / 2);

                ctx.fillStyle = "white";

                ctx.font = "12px 'Noto Sans TC', sans-serif";

                ctx.textAlign = "center";

                ctx.fillText(keywords[i], 60, 5);

                ctx.restore();

            }

        }


        let rotation = 0;

        function spinWheel() {

            const randomDegree = Math.floor(Math.random() * 360) + 1440; // 讓輪盤至少轉 4 圈

            rotation += randomDegree;

            canvas.style.transition = 'transform 3s ease-out';

            canvas.style.transform = `rotate(${rotation}deg)`;


            setTimeout(() => {

                const actualDegree = (rotation % 360); // 計算最終角度

                let correctedDegree = (360 - actualDegree + 270) % 360; // 修正對應角度 (+270 讓指針對應 12 點方向)


                // 計算正確的區塊索引

                const index = Math.floor(correctedDegree / (360 / keywords.length));

                resultText.innerHTML = `<div style='font-size:24px; font-weight:bold; color:#D32F2F;'>🎉 你抽中了：${keywords[index]}！ 🎉</div>

                    <div style='font-size:18px; margin-top:5px;'>你的新年智慧訊息已準備好，請私訊 @originpaths 領取你的專屬指引與驚喜！ ✨💫</div>`;

            }, 3000);

        }


        spinButton.addEventListener("click", spinWheel);

        drawWheel();
